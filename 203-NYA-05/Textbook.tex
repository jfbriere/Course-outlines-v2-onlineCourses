\entry{Required materials}{
% FALL + WINTER
\begin{enumerate}
	{\color{red}
	\item Exceptionally you are required to have a webcam or a camera (can be the one from a phone).
	\item Exceptionally you are required to have access to internet to follow the course.
	}
	\item Your teacher will tell you which {\bf one of the two textbooks} will be used in your section and whether you need an access code or not for the online homework system.
	\begin{itemize}
		\item {\bf Knight, Physics for Scientists and Engineers (with Mastering Physics), 4th edition}, no hard copy available, only e-book available through the bookstore; \\ or
		\item {\bf Serway custom package for Dawson College NYA} containing excerpts from {\it Physics for Scientists and Engineers (with Enhanced WebAssign) by Serway \& Jewett, 9th edition}, e-book and hard copy available through the bookstore.\\
		The custom packages available through the bookstore include a semester-long access code for the online homework system. Used textbook generally do not include an access code.
	\end{itemize}
	\item {\bf Library copies:} Copies of the textbooks are available on reserve in the Dawson Library (as are similar textbooks)
\end{enumerate}
}
